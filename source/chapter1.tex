反転増幅器\index{はんてんぞうふくき@反転増幅器}\\
\begin{align}
\begin{split}
v_{in} &= R_{1} i \\
\end{split}
\end{align}

\begin{align}
\begin{split}
v_{in} &=( R_{1} + R_{2}) i + v_{out} \\
v_{in} &=( R_{1} + R_{2})  \times \frac{v_{in}}{R_{1}}+ v_{out} \\
v_{in} &= v_{in} + \frac{R_{2}}{R_{1}}) v_{in}  + v_{out} \\
\frac{v_{out}}{v_{in}} &= -\frac{R_{2}}{R_{1}}\\
\end{split}
\end{align}

非反転増幅器\index{おぺあんぷ@オペアンプ!ひはんてんぞうふくき@非反転増幅器}\\
\begin{align}
\begin{split}
v_{in} &= R_{1} i \\
\end{split}
\end{align}

\begin{align}
\begin{split}
v_{in} + R_{2} i &=  v_{out} \\
v_{in} + R_{2} \times \frac{v_{in}}{R_{1}} &= v_{out} \\
\frac{R_{1}+R_{2}}{R_{1}} v_{in} &= v_{out} \\
\frac{v_{out}}{v_{in}} &= \frac{R_{1}+R_{2}}{R_{1}}\\
\end{split}
\end{align}
