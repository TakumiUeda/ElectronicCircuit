反転増幅器\index{はんてんぞうふくき@反転増幅器}\\
\begin{align}
\begin{split}
v_{in} &= R_{1} i \\
\end{split}
\end{align}

\begin{align}
\begin{split}
v_{in} &=( R_{1} + R_{2}) i + v_{out} \\
v_{in} &=( R_{1} + R_{2})  \times \frac{v_{in}}{R_{1}}+ v_{out} \\
v_{in} &= \left( v_{in} + \frac{R_{2}}{R_{1}} \right ) v_{in}  + v_{out} \\
\frac{v_{out}}{v_{in}} &= -\frac{R_{2}}{R_{1}}\\
\end{split}
\end{align}

非反転増幅器\index{ひはんてんぞうふくき@非反転増幅器}\\
\begin{align}
\begin{split}
v_{in} &= R_{1} i \\
\end{split}
\end{align}

\begin{align}
\begin{split}
v_{in} + R_{2} i &=  v_{out} \\
v_{in} + R_{2} \times \frac{v_{in}}{R_{1}} &= v_{out} \\
\frac{R_{1}+R_{2}}{R_{1}} v_{in} &= v_{out} \\
\frac{v_{out}}{v_{in}} &= \frac{R_{1}+R_{2}}{R_{1}}\\
\end{split}
\end{align}
ボルテージフォロア\index{ぼるてーじふぉろあ@ボルテージフォロア}\\
非反転増幅器の特性
\begin{align}
\begin{split}
\frac{v_{out}}{v_{in}} &= \frac{R_{1}+R_{2}}{R_{1}}\\
\end{split}
\end{align}
において，$R_{2} = 0,R_{1} = \infty$とすれば
\begin{align}
\begin{split}
v_{out} &= v_{in}\\
\end{split}
\end{align}
